async function gerarDesignIA() { const prompt = document.getElementById('userPrompt').value; const segmento = document.getElementById('segmento').value; const estilo = document.getElementById('estilo').value; const imgElement = document.getElementById('outputImage'); if(!prompt) return alert("Por favor, descreva a estampa!"); alert('Processando estampa para: ' + segmento + '...'); try { const response = await fetch('https://api.openai.com/v1/images/generations', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${API_KEY}` }, body: JSON.stringify({ model: "dall-e-3", prompt: `Estampa seamless para ${segmento}, estilo ${estilo}. Tema: ${prompt}. Alta definição, profissional.`, n: 1, size: "1024x1024" }) }); const data = await response.json(); if(data.data && data.data[0].url) { imgElement.src = data.data[0].url; imgElement.style.display = 'block'; } else { alert('Erro ao gerar: Verifique sua chave API e saldo na OpenAI.'); } } catch (e) { alert('Erro no Motor. Verifique sua conexão.'); } }
